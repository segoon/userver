124  /// @brief Sets the limit on the queue size, pushes over this limit will block
125  /// @note This is a soft limit and may be slightly overrun under load.
126  /// @note If the current queue size is greater than new max size, this call
127  /// will block until size becomes less than the new max size
128  void SetSoftMaxSize(std::size_t max_size) {
129  max_size = std::min(max_size, kUnbounded);
130  const auto old_capacity = capacity_.exchange(max_size);
131  if (max_size > old_capacity) {
132  producer_side_.IncreaseCapacity(max_size - old_capacity);
133  } else if (max_size < old_capacity) {
134  producer_side_.DecreaseCapacity(old_capacity - max_size);
135  }
136  }

202  void DoPush(ProducerToken& token, T&& value) {
203  if constexpr (MultipleProducer) {
204  queue_.enqueue(token, std::move(value));
205  } else {
206  // If producer is single substitute his token
207  queue_.enqueue(single_producer_token_, std::move(value));
208  }
209 
210  consumer_side_.OnElementPushed();
211  }
212 
213  [[nodiscard]] bool DoPop(ConsumerToken& token, T& value) {
214  bool success = false;
215  if constexpr (MultipleProducer) {
216  success = queue_.try_dequeue(token, value);
217  } else {
218  // Substitute with our single producer token
219  success = queue_.try_dequeue_from_producer(single_producer_token_, value);
220  }
221 
222  if (success) {
223  producer_side_.OnElementPopped();
224  return true;
225  }
226 
227  return false;
228  }

247 /// Single producer ProducerSide implementation
248 template <typename T, bool MP, bool MC>
249 class GenericQueue<T, MP, MC>::SingleProducerSide final {
250  public:
251  explicit SingleProducerSide(GenericQueue& queue)
252  : queue_(queue), remaining_capacity_(0) {}
253 
254  /// Blocks if there is a consumer to Pop the current value and task
255  /// shouldn't cancel and queue if full
256  [[nodiscard]] bool Push(ProducerToken& token, T&& value,
257  engine::Deadline deadline) {
258  if (DoPush(token, std::move(value))) {
259  return true;
260  }
261 
262  return nonfull_event_.WaitForEventUntil(deadline) &&
263  DoPush(token, std::move(value));
264  }
265 
266  [[nodiscard]] bool PushNoblock(ProducerToken& token, T&& value) {
267  return DoPush(token, std::move(value));
268  }
269 
270  void OnElementPopped() {
271  ++remaining_capacity_;
272  nonfull_event_.Send();
273  }
274 
275  void DecreaseCapacity(std::size_t count) { remaining_capacity_ -= count; }
276 
277  void IncreaseCapacity(std::size_t count) {
278  remaining_capacity_ += count;
279  nonfull_event_.Send();
280  }
281 
282  private:
283  [[nodiscard]] bool DoPush(ProducerToken& token, T&& value) {
284  if (queue_.NoMoreConsumers() || !remaining_capacity_) {
285  return false;
286  }
287 
288  --remaining_capacity_;
289  queue_.DoPush(token, std::move(value));
290  nonfull_event_.Reset();
291  return true;
292  }
293 
294  GenericQueue& queue_;
295  engine::SingleConsumerEvent nonfull_event_;
296  std::atomic<std::size_t> remaining_capacity_;
297 };

353 /// Single consumer ConsumerSide implementation
354 template <typename T, bool MP, bool MC>
355 class GenericQueue<T, MP, MC>::SingleConsumerSide final {
356  public:
357  explicit SingleConsumerSide(GenericQueue& queue) : queue_(queue), size_(0) {}
358 
359  /// Blocks only if queue is empty
360  [[nodiscard]] bool Pop(ConsumerToken& token, T& value,
361  engine::Deadline deadline) {
362  while (!DoPop(token, value)) {
363  if (queue_.NoMoreProducers() ||
364  !nonempty_event_.WaitForEventUntil(deadline)) {
365  // Producer might have pushed something in queue between .pop()
366  // and !producer_is_created_and_dead_ check. Check twice to avoid
367  // TOCTOU.
368  return DoPop(token, value);
369  }
370  }
371  return true;
372  }
373 
374  [[nodiscard]] bool PopNoblock(ConsumerToken& token, T& value) {
375  return DoPop(token, value);
376  }
377 
378  void OnElementPushed() {
379  ++size_;
380  nonempty_event_.Send();
381  }
382 
383  void StopBlockingOnPop() { nonempty_event_.Send(); }
384 
385  void ResumeBlockingOnPop() {}
386 
387  std::size_t GetSize() const { return size_; }
388 
389  private:
390  [[nodiscard]] bool DoPop(ConsumerToken& token, T& value) {
391  if (queue_.DoPop(token, value)) {
392  --size_;
393  nonempty_event_.Reset();
394  return true;
395  }
396  return false;
397  }
398 
399  GenericQueue& queue_;
400  engine::SingleConsumerEvent nonempty_event_;
401  std::atomic<std::size_t> size_;
