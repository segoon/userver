84 Read(Reader& reader, To<T>) {

196 template <typename L, typename R, typename AP1, typename AP2, typename AP3>
197 std::enable_if_t<kIsWritable<L> && kIsWritable<R>,
198  boost::bimap<L, R, AP1, AP2, AP3>>
199 Read(Reader& reader, To<boost::bimap<L, R, AP1, AP2, AP3>>) {
200  const auto size = reader.Read<std::size_t>();
201 
202  boost::bimap<L, R, AP1, AP2, AP3> map;
203  // bimap doesn't have reserve :(
204 
205  for (std::size_t i = 0; i < size; ++i) {
206  // `Read`s are guaranteed to occur left-to-right in brace-init
