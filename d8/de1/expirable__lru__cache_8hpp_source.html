311 template <typename Key, typename Value, typename Hash, typename Equal>
312 void ExpirableLruCache<Key, Value, Hash, Equal>::UpdateInBackground(
313  const Key& key, UpdateValueFunc update_func) {
314  stats_.total.background_updates++;
315  stats_.recent.GetCurrentCounter().background_updates++;
316 
317  // cache will wait for all detached tasks in ~ExpirableLruCache()
318  engine::AsyncNoSpan([token = wait_token_storage_.GetToken(), this, key,
319  update_func = std::move(update_func)] {
320  auto mutex = mutex_set_.GetMutexForKey(key);
321  std::unique_lock lock(mutex, std::try_to_lock);
322  if (!lock) {
323  // someone is updating the key right now
324  return;
325  }
326 
327  auto now = utils::datetime::SteadyNow();
328  auto value = update_func(key);
329  lru_.Put(key, {value, now});
330  }).Detach();
331 }
