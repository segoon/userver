89 template <int MaxExp>
90 constexpr std::array<int64_t, MaxExp + 1> PowSeries(int64_t base) {
91  int64_t pow = 1;
92  std::array<int64_t, MaxExp + 1> result{};
93  for (int i = 0; i < MaxExp; ++i) {
94  result[i] = pow;
95  if (pow > kMaxInt64 / base) {
96  throw OutOfBoundsError();
97  }
98  pow *= base;
99  }
100  result[MaxExp] = pow;
101  return result;
102 }

123 // result = (value1 * value2) / divisor
124 template <typename RoundPolicy>
125 constexpr int64_t MulDiv(int64_t value1, int64_t value2, int64_t divisor) {
126  if (divisor == 0) throw DivisionByZeroError();
127 
128 #if __x86_64__ || __ppc64__
129  using LongInt = __int128_t;
130  static_assert(sizeof(void*) == 8);
131 #else
132  using LongInt = int64_t;
133  static_assert(sizeof(void*) == 4);
134 #endif
135 
136  LongInt prod{};
137  if constexpr (sizeof(void*) == 4) {
138  if (__builtin_mul_overflow(static_cast<LongInt>(value1),
139  static_cast<LongInt>(value2), &prod)) {
140  throw OutOfBoundsError();
141  }
142  } else {
143  prod = static_cast<LongInt>(value1) * value2;
144  }
145  const auto whole = prod / divisor;
146  const auto rem = static_cast<int64_t>(prod % divisor);
147 
148  if (whole <= kMinInt64 || whole >= kMaxInt64) throw OutOfBoundsError();
149 
150  const auto whole64 = static_cast<int64_t>(whole);
151  const bool extra_odd_quotient = whole64 % 2 != 0;
152  const int64_t rem_divided =
153  Div<RoundPolicy>(rem, divisor, extra_odd_quotient);
154  UASSERT(rem_divided == -1 || rem_divided == 0 || rem_divided == 1);
155 
156  return whole64 + rem_divided;
157 }

263  const int64_t half_b = abs_b / 2;

308 /// @brief Default rounding. Fast, rounds to nearest.
309 ///
310 /// On 0.5, rounds away from zero. Also, sometimes rounds up numbers
311 /// in the neighborhood of 0.5, e.g. 0.49999999999999994 -> 1.
312 class DefRoundPolicy final {
313  public:
314  template <typename T>
315  [[nodiscard]] static constexpr int64_t Round(T value) {
316  return static_cast<int64_t>(value + (value < 0 ? -0.5 : 0.5));
317  }
318 
319  [[nodiscard]] static constexpr int64_t DivRounded(
320  int64_t a, int64_t b, bool /*extra_odd_quotient*/) {
321  const int64_t divisor_corr = impl::Abs(b / 2);
322  if (a >= 0) {
323  if (impl::kMaxInt64 - a < divisor_corr) throw OutOfBoundsError();
324  return (a + divisor_corr) / b;
325  } else {
326  if (-(impl::kMinInt64 - a) < divisor_corr) throw OutOfBoundsError();
327  return (a - divisor_corr) / b;
328  }
329  }
330 };

407 /// @ingroup userver_containers
408 ///
409 /// @brief Fixed-point decimal data type for use in deterministic calculations,
410 /// oftentimes involving money
411 ///
412 /// @tparam Prec The number of fractional digits
413 /// @tparam RoundPolicy Specifies how to round in lossy operations
414 ///
415 /// Decimal is internally represented as `int64_t`. It means that it can be
416 /// passed around by value. It also means that operations with huge
417 /// numbers can overflow and trap. For example, with `Prec == 6`, the maximum
418 /// representable number is about 10 trillion.
419 ///
420 /// Decimal should be serialized and stored as a string, NOT as `double`. Use
421 /// `Decimal{str}` constructor (or `Decimal::FromStringPermissive` if rounding
422 /// is allowed) to read a `Decimal`, and `ToString(dec)`
423 /// (or `ToStringTrailingZeros(dec)`/`ToStringFixed<N>(dec)`) to write a
424 /// `Decimal`.
425 ///
426 /// Use arithmetic with caution! Multiplication and division operations involve
427 /// rounding. You may want to cast to `Decimal` with another `Prec`
428 /// or `RoundPolicy` beforehand. For that purpose you can use
429 /// `decimal64::decimal_cast<NewDec>(dec)`.
430 ///
431 /// Usage example:
432 /// @code{.cpp}
433 /// // create a single alias instead of specifying Decimal everywhere
434 /// using Money = decimal64::Decimal<4, decimal64::HalfEvenRoundPolicy>;
435 ///
436 /// std::vector<std::string> cart = ...;
437 /// Money sum{0};
438 /// for (const std::string& cost_string : cart) {
439 /// // or use FromStringPermissive to enable rounding
440 /// sum += Money{cost_string};
441 /// }
442 /// return ToString(sum);
443 /// @endcode

479  ///
480  /// To somewhat resist the accumulated error, the number is always rounded
481  /// to the nearest Decimal, regardless of `RoundPolicy`.
482  ///
483  /// @warning Prefer storing and sending `Decimal` as string, and performing
484  /// the computations between `Decimal`s.

520  /// @brief Convert from `original_unbiased / 10^original_precision`, rounding
521  /// according to `RoundPolicy` if necessary
522  ///
523  /// Usage examples:
524  ///
525  /// Decimal<4>::FromBiased(123, 6) -> 0.0001
526  /// Decimal<4>::FromBiased(123, 2) -> 1.23
527  /// Decimal<4>::FromBiased(123, -1) -> 1230
528  ///
529  /// @param original_unbiased The original mantissa
530  /// @param original_precision The original precision (negated exponent)
531  static constexpr Decimal FromBiased(int64_t original_unbiased,
532  int original_precision) {

534 
535  if (exponent_for_pack >= 0) {
536  return FromUnbiased(original_unbiased) * Pow10(exponent_for_pack);

539  impl::Div<RoundPolicy>(original_unbiased, Pow10(-exponent_for_pack)));
540  }

817 // FromUnpacked<Decimal<4>>(12, 34) -> 12.0034
818 // FromUnpacked<Decimal<4>>(-12, -34) -> -12.0034
819 // FromUnpacked<Decimal<4>>(0, -34) -> -0.0034
820 template <int Prec, typename RoundPolicy>
821 constexpr Decimal<Prec, RoundPolicy> FromUnpacked(int64_t before,
822  int64_t after) {
823  using Dec = Decimal<Prec, RoundPolicy>;
824  UASSERT(((before >= 0) && (after >= 0)) || ((before <= 0) && (after <= 0)));
825 
826  int64_t result{};
827  if (__builtin_mul_overflow(before, Dec::kDecimalFactor, &result) ||
828  __builtin_add_overflow(result, after, &result)) {
829  throw OutOfBoundsError();
830  }
831 
832  return Dec::FromUnbiased(result);
833 }

835 // FromUnpacked<Decimal<4>>(12, 34, 3) -> 12.034
836 template <int Prec, typename RoundPolicy>
837 constexpr Decimal<Prec, RoundPolicy> FromUnpacked(int64_t before, int64_t after,
838  int original_precision) {
839  UASSERT(((before >= 0) && (after >= 0)) || ((before <= 0) && (after <= 0)));
840  UASSERT(after > -Pow10(original_precision) &&
841  after < Pow10(original_precision));
842 
843  if (original_precision <= Prec) {
844  // direct mode
845  const int missing_digits = Prec - original_precision;
846  const int64_t factor = Pow10(missing_digits);
847  return FromUnpacked<Prec, RoundPolicy>(before, after * factor);
848  } else {
849  // rounding mode
850  const int extra_digits = original_precision - Prec;
851  const int64_t factor = Pow10(extra_digits);
852  // note: if rounded up, rounded_after may represent a "fractional part"
853  // greater than 1.0, which is ok
854  const int64_t rounded_after = Div<RoundPolicy>(after, factor);
855  return FromUnpacked<Prec, RoundPolicy>(before, rounded_after);
856  }
857 }

864 // AsUnpacked(Decimal<4>{"3.14"}) -> {3, 1400}
865 // AsUnpacked(Decimal<4>{"-3.14"}) -> {-3, -1400}
866 // AsUnpacked(Decimal<4>{"-0.14"}) -> {0, -1400}
867 template <int Prec, typename RoundPolicy>
868 constexpr UnpackedDecimal AsUnpacked(Decimal<Prec, RoundPolicy> dec) {
869  using Dec = Decimal<Prec, RoundPolicy>;
870  return {dec.AsUnbiased() / Dec::kDecimalFactor,
871  dec.AsUnbiased() % Dec::kDecimalFactor};
872 }
873 
874 // AsUnpacked(Decimal<4>{"3.14"}, 5) -> {3, 14000}
875 // AsUnpacked(Decimal<4>{"-3.14"}, 6) -> {-3, -140000}
876 // AsUnpacked(Decimal<4>{"-0.14"}, 1) -> {0, -1}
877 template <int Prec, typename RoundPolicy>
878 UnpackedDecimal AsUnpacked(Decimal<Prec, RoundPolicy> dec, int new_prec) {
879  if (new_prec == Prec) {
880  return AsUnpacked(dec);
881  }
882  int64_t result{};
883  if (new_prec > Prec) {
884  if (__builtin_mul_overflow(dec.AsUnbiased(), Pow10(new_prec - Prec),
885  &result)) {
886  throw OutOfBoundsError();
887  }
888  } else {
889  result = impl::Div<RoundPolicy>(dec.AsUnbiased(), Pow10(Prec - new_prec));
890  }
891  const auto dec_factor = Pow10(new_prec);
892  return {result / dec_factor, result % dec_factor};
893 }

1020 /// Extract values from a CharSequence ready to be packed to Decimal
1021 template <typename CharSequence>
1022 [[nodiscard]] constexpr ParseUnpackedResult ParseUnpacked(
1023  CharSequence input, utils::Flags<ParseOptions> options) {
1024  constexpr char dec_point = '.';
1025 
1026  int64_t before = 0;
1027  int64_t after = 0;
1028  bool is_negative = false;
1029 
1030  ptrdiff_t position = -1;
1031  auto state = ParseState::kSign;
1032  std::optional<ParseErrorCode> error;
1033  int before_digit_count = 0;
1034  uint8_t after_digit_count = 0;
1035 
1036  while (state != ParseState::kEnd) {
1037  const auto c = input.Get();
1038  if (c == '\0') break;
1039  if (!error) ++position;
1040 
1041  switch (state) {
1042  case ParseState::kSign:
1043  if (c == '-') {
1044  is_negative = true;
1045  state = ParseState::kBeforeFirstDig;
1046  } else if (c == '+') {
1047  state = ParseState::kBeforeFirstDig;
1048  } else if (c == '0') {
1049  state = ParseState::kLeadingZeros;
1050  before_digit_count = 1;
1051  } else if ((c >= '1') && (c <= '9')) {
1052  state = ParseState::kBeforeDec;
1053  before = static_cast<int>(c - '0');
1054  before_digit_count = 1;
1055  } else if (c == dec_point) {
1056  if (!(options & ParseOptions::kAllowBoundaryDot) && !error) {
1057  error = ParseErrorCode::kBoundaryDot; // keep reading digits
1058  }
1059  state = ParseState::kAfterDec;
1060  } else if (IsSpace(c)) {
1061  if (!(options & ParseOptions::kAllowSpaces)) {
1062  state = ParseState::kEnd;
1063  error = ParseErrorCode::kSpace;
1064  }
1065  } else {
1066  state = ParseState::kEnd;
1067  error = ParseErrorCode::kWrongChar;
1068  }
1069  break;
1070  case ParseState::kBeforeFirstDig:
1071  if (c == '0') {
1072  state = ParseState::kLeadingZeros;
1073  before_digit_count = 1;
1074  } else if ((c >= '1') && (c <= '9')) {
1075  state = ParseState::kBeforeDec;
1076  before = static_cast<int>(c - '0');
1077  before_digit_count = 1;
1078  } else if (c == dec_point) {
1079  if (!(options & ParseOptions::kAllowBoundaryDot) && !error) {
1080  error = ParseErrorCode::kBoundaryDot; // keep reading digits
1081  }
1082  state = ParseState::kAfterDec;
1083  } else {
1084  state = ParseState::kEnd;
1085  error = ParseErrorCode::kWrongChar;
1086  }
1087  break;
1088  case ParseState::kLeadingZeros:
1089  if (c == '0') {
1090  // skip
1091  } else if ((c >= '1') && (c <= '9')) {
1092  state = ParseState::kBeforeDec;
1093  before = static_cast<int>(c - '0');
1094  } else if (c == dec_point) {
1095  state = ParseState::kAfterDec;
1096  } else {
1097  state = ParseState::kEnd;
1098  }
1099  break;
1100  case ParseState::kBeforeDec:
1101  if ((c >= '0') && (c <= '9')) {
1102  if (before_digit_count < kMaxDecimalDigits) {
1103  before = 10 * before + static_cast<int>(c - '0');
1104  before_digit_count++;
1105  } else if (!error) {
1106  error = ParseErrorCode::kOverflow; // keep reading digits
1107  }
1108  } else if (c == dec_point) {
1109  state = ParseState::kAfterDec;
1110  } else {
1111  state = ParseState::kEnd;
1112  }
1113  break;
1114  case ParseState::kAfterDec:
1115  if ((c >= '0') && (c <= '9')) {
1116  if (after_digit_count < kMaxDecimalDigits) {
1117  after = 10 * after + static_cast<int>(c - '0');
1118  after_digit_count++;
1119  } else {
1120  if (!(options & ParseOptions::kAllowRounding) && !error) {
1121  error = ParseErrorCode::kRounding; // keep reading digits
1122  }
1123  state = ParseState::kIgnoringAfterDec;
1124  if (c >= '5') {
1125  // round half up
1126  after++;
1127  }
1128  }
1129  } else {
1130  if (!(options & ParseOptions::kAllowBoundaryDot) &&
1131  after_digit_count == 0 && !error) {
1132  error = ParseErrorCode::kBoundaryDot;
1133  }
1134  state = ParseState::kEnd;
1135  }
1136  break;
1137  case ParseState::kIgnoringAfterDec:
1138  if ((c >= '0') && (c <= '9')) {
1139  // skip
1140  } else {
1141  state = ParseState::kEnd;
1142  }
1143  break;
1144  case ParseState::kEnd:
1145  UASSERT(false);
1146  break;
1147  } // switch state
1148  } // while has more chars & not end
1149 
1150  if (state == ParseState::kEnd) {
1151  input.Unget();
1152 
1153  if (!error && !(options & ParseOptions::kAllowTrailingJunk)) {
1154  if (!(options & ParseOptions::kAllowSpaces)) {
1155  error = ParseErrorCode::kSpace;
1156  }
1157  --position;
1158 
1159  while (true) {
1160  const auto c = input.Get();
1161  if (c == '\0') break;
1162  ++position;
1163  if (!IsSpace(c)) {
1164  error = ParseErrorCode::kTrailingJunk;
1165  input.Unget();
1166  break;
1167  }
1168  }
1169  }
1170  }
1171 
1172  if (!error && before_digit_count == 0 && after_digit_count == 0) {
1173  error = ParseErrorCode::kNoDigits;
1174  }
1175 
1176  if (!error && state == ParseState::kAfterDec &&
1177  !(options & ParseOptions::kAllowBoundaryDot) && after_digit_count == 0) {
1178  error = ParseErrorCode::kBoundaryDot;
1179  }
1180 
1181  return {before, after, after_digit_count,
1182  is_negative, error, static_cast<uint32_t>(position)};
1183 }
